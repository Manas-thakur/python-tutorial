## Project 1: Async API Client

Build an async API client that fetches data from multiple endpoints concurrently, handles rate limiting, and includes retry logic.

### Features
- Fetch multiple API endpoints concurrently using `asyncio` + `aiohttp`
- Rate limiting with a semaphore-based throttle
- Retry decorator for transient failures
- Response caching with TTL (time-to-live)
- Command-line interface for selecting endpoints

### Code

```python
import asyncio
import aiohttp
import time
import json
import functools
from dataclasses import dataclass, field
from typing import Any, Optional

# --- Retry decorator ---
def retry(max_attempts=3, delay=1, backoff=2):
    def decorator(func):
        @functools.wraps(func)
        async def wrapper(*args, **kwargs):
            current_delay = delay
            for attempt in range(1, max_attempts + 1):
                try:
                    return await func(*args, **kwargs)
                except (aiohttp.ClientError, asyncio.TimeoutError) as e:
                    print(f"Attempt {attempt}/{max_attempts} failed: {e}")
                    if attempt == max_attempts:
                        raise
                    await asyncio.sleep(current_delay)
                    current_delay *= backoff
            return None
        return wrapper
    return decorator

# --- Rate limiter ---
class RateLimiter:
    def __init__(self, max_concurrent: int):
        self.semaphore = asyncio.Semaphore(max_concurrent)

    async def __aenter__(self):
        await self.semaphore.acquire()

    async def __aexit__(self, *args):
        self.semaphore.release()

# --- Caching ---
@dataclass
class CacheEntry:
    data: Any
    timestamp: float

class TTLCache:
    def __init__(self, ttl_seconds: int = 60):
        self.cache: dict[str, CacheEntry] = {}
        self.ttl = ttl_seconds

    def get(self, key: str) -> Optional[Any]:
        entry = self.cache.get(key)
        if entry and (time.time() - entry.timestamp) < self.ttl:
            return entry.data
        return None

    def set(self, key: str, data: Any):
        self.cache[key] = CacheEntry(data, time.time())

    def invalidate(self, key: str):
        self.cache.pop(key, None)

# --- API Client ---
class AsyncAPIClient:
    def __init__(self, max_concurrent: int = 5, ttl: int = 60):
        self.rate_limiter = RateLimiter(max_concurrent)
        self.cache = TTLCache(ttl)
        self.session: Optional[aiohttp.ClientSession] = None

    async def __aenter__(self):
        self.session = aiohttp.ClientSession(
            timeout=aiohttp.ClientTimeout(total=30)
        )
        return self

    async def __aexit__(self, *args):
        if self.session:
            await self.session.close()

    @retry(max_attempts=3, delay=1)
    async def fetch(self, url: str, use_cache: bool = True) -> Any:
        if use_cache:
            cached = self.cache.get(url)
            if cached:
                print(f"[CACHE] HIT for {url}")
                return cached

        async with self.rate_limiter:
            print(f"[FETCH] {url}")
            async with self.session.get(url) as response:
                response.raise_for_status()
                data = await response.json()

        if use_cache:
            self.cache.set(url, data)
        return data

    async def fetch_many(self, urls: list[str], use_cache: bool = True) -> list[Any]:
        tasks = [self.fetch(url, use_cache) for url in urls]
        return await asyncio.gather(*tasks, return_exceptions=True)

# --- Demo ---
async def main():
    endpoints = {
        "jsonplaceholder": "https://jsonplaceholder.typicode.com/posts/",
        "httpbin delay": "https://httpbin.org/delay/1",
        "httpbin uuid": "https://httpbin.org/uuid",
    }

    async with AsyncAPIClient(max_concurrent=3, ttl=120) as client:
        # Fetch all endpoints concurrently
        urls = list(endpoints.values())
        print("=== First fetch (no cache) ===")
        results = await client.fetch_many(urls)

        for name, result in zip(endpoints.keys(), results):
            if isinstance(result, Exception):
                print(f"[ERROR] {name}: {result}")
            elif isinstance(result, list):
                print(f"[{name}] Got {len(result)} items")
            else:
                print(f"[{name}] Got response")

        print("\n=== Second fetch (cached) ===")
        results2 = await client.fetch_many(urls)

        # Individual fetches with different URLs
        print("\n=== Individual fetch ===")
        post = await client.fetch("https://jsonplaceholder.typicode.com/posts/1")
        print(f"Post title: {post.get('title', 'N/A')}")

if __name__ == "__main__":
    asyncio.run(main())
```

### Key concepts demonstrated
- `async with` for session management
- `asyncio.gather()` for concurrent requests
- Decorator factory (`@retry`) with async wrapper
- Context manager (`RateLimiter`) with async enter/exit
- TTL caching with cache hit logging
- `return_exceptions=True` to handle partial failures

## Project 2: Async Web Scraper

Build an async web scraper that fetches multiple URLs concurrently, parses HTML with BeautifulSoup, extracts data, and saves results to a file.

### Features
- Concurrent page fetching with `aiohttp`
- HTML parsing with `BeautifulSoup`
- Extracts: title, meta description, all links, images
- Saves results to JSON
- Graceful error handling (bad URLs, timeouts, parse errors)
- Rate limiting to be polite to servers

### Code

```python
import asyncio
import aiohttp
from bs4 import BeautifulSoup
import json
import time
import functools
from urllib.parse import urljoin, urlparse

# --- Rate limiter (per-domain) ---
class DomainRateLimiter:
    def __init__(self, requests_per_second: float = 2.0):
        self.delays: dict[str, float] = {}
        self.min_interval = 1.0 / requests_per_second

    async def wait(self, url: str):
        domain = urlparse(url).netloc
        now = time.time()
        last = self.delays.get(domain, 0)
        wait_time = last + self.min_interval - now
        if wait_time > 0:
            await asyncio.sleep(wait_time)
        self.delays[domain] = time.time()

# --- Async Web Scraper ---
class AsyncWebScraper:
    def __init__(self, max_concurrent: int = 5, requests_per_second: float = 3.0):
        self.semaphore = asyncio.Semaphore(max_concurrent)
        self.rate_limiter = DomainRateLimiter(requests_per_second)

    async def fetch_page(self, session: aiohttp.ClientSession, url: str) -> str:
        await self.rate_limiter.wait(url)
        async with self.semaphore:
            async with session.get(url, timeout=aiohttp.ClientTimeout(total=15)) as resp:
                resp.raise_for_status()
                return await resp.text()

    def parse_page(self, url: str, html: str) -> dict:
        soup = BeautifulSoup(html, "html.parser")

        # Title
        title = soup.title.string.strip() if soup.title else ""

        # Meta description
        meta_desc = ""
        meta_tag = soup.find("meta", attrs={"name": "description"})
        if meta_tag:
            meta_desc = meta_tag.get("content", "")

        # All links
        links = []
        for a_tag in soup.find_all("a", href=True):
            href = urljoin(url, a_tag["href"])
            text = a_tag.get_text(strip=True)
            links.append({"href": href, "text": text or ""})

        # Images
        images = []
        for img in soup.find_all("img", src=True):
            src = urljoin(url, img["src"])
            alt = img.get("alt", "")
            images.append({"src": src, "alt": alt})

        # Headings
        headings = {}
        for level in range(1, 4):
            tag = f"h{level}"
            headings[tag] = [h.get_text(strip=True) for h in soup.find_all(tag)]

        return {
            "url": url,
            "title": title,
            "meta_description": meta_desc,
            "links_count": len(links),
            "images_count": len(images),
            "links": links[:20],         # Limit to 20 links
            "images": images[:10],       # Limit to 10 images
            "headings": headings,
        }

    async def scrape(self, session: aiohttp.ClientSession, url: str) -> dict:
        try:
            html = await self.fetch_page(session, url)
            data = self.parse_page(url, html)
            print(f"[OK] {url} — {data['title'][:60]}")
            return data
        except Exception as e:
            print(f"[FAIL] {url} — {type(e).__name__}: {e}")
            return {"url": url, "error": str(e)}

    async def scrape_many(self, urls: list[str]) -> list[dict]:
        async with aiohttp.ClientSession() as session:
            tasks = [self.scrape(session, url) for url in urls]
            results = await asyncio.gather(*tasks)
            return results

# --- Main ---
async def main():
    urls = [
        "https://httpbin.org/html",
        "https://example.com",
        "https://www.python.org",
        "https://github.com",
        "https://news.ycombinator.com",
    ]

    scraper = AsyncWebScraper(max_concurrent=3, requests_per_second=2)
    print(f"Scraping {len(urls)} URLs...\n")
    start = time.perf_counter()

    results = await scraper.scrape_many(urls)

    elapsed = time.perf_counter() - start
    successes = sum(1 for r in results if "error" not in r)

    print(f"\n--- Summary ---")
    print(f"Total: {len(urls)} | Success: {successes} | Failed: {len(urls) - successes}")
    print(f"Time: {elapsed:.2f}s")

    # Save to JSON
    with open("scrape_results.json", "w", encoding="utf-8") as f:
        json.dump(results, f, indent=2, ensure_ascii=False)
    print("Results saved to scrape_results.json")

    # Pretty print first successful result
    for r in results:
        if "error" not in r:
            print(f"\nSample: {r['url']}")
            print(f"  Title: {r['title']}")
            print(f"  Links: {r['links_count']}")
            print(f"  Images: {r['images_count']}")
            break

if __name__ == "__main__":
    asyncio.run(main())
```

### Key concepts demonstrated
- Per-domain rate limiting (polite scraping)
- `asyncio.Semaphore` for concurrency control
- `BeautifulSoup` for HTML parsing
- `urljoin` for resolving relative URLs
- `asyncio.gather` for concurrent scraping
- Results serialization to JSON
- Error handling per-URL (partial failures don't stop others)

## Project 3: Concurrent Downloader

Build a concurrent file downloader that downloads multiple files using `ThreadPoolExecutor` (I/O-bound) and `ProcessPoolExecutor` (CPU-bound processing like hashing/resizing), with progress tracking.

### Features
- Download multiple files concurrently using threads
- Optional CPU-bound post-processing (hash verification, image resize)
- Progress bar with `tqdm`
- Resume support (partial download -> HTTP Range header)
- Configurable: thread pool size, process pool size, output directory
- Error handling with retry

### Code

```python
import os
import hashlib
import time
import functools
import requests
from concurrent.futures import ThreadPoolExecutor, ProcessPoolExecutor, as_completed
from dataclasses import dataclass, field
from typing import Optional
from pathlib import Path

try:
    from tqdm import tqdm
except ImportError:
    tqdm = None

# --- Retry decorator ---
def retry(max_attempts=3, delay=1):
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(1, max_attempts + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    print(f"  Retry {attempt}/{max_attempts} after {e}")
                    if attempt == max_attempts:
                        raise
                    time.sleep(delay)
            return None
        return wrapper
    return decorator

# --- Download item ---
@dataclass
class DownloadItem:
    url: str
    filename: str
    size_bytes: int = 0
    status: str = "pending"
    error: Optional[str] = None
    sha256: Optional[str] = None

# --- Download function (I/O-bound, threading) ---
@retry(max_attempts=3, delay=1)
def download_file(url: str, output_dir: Path, expected_hash: Optional[str] = None) -> DownloadItem:
    local_filename = url.split("/")[-1] or "index.html"
    output_path = output_dir / local_filename

    # Check if partially downloaded
    headers = {}
    resume_bytes = 0
    if output_path.exists():
        resume_bytes = output_path.stat().st_size
        headers["Range"] = f"bytes={resume_bytes}-"

    resp = requests.get(url, headers=headers, stream=True, timeout=30)
    resp.raise_for_status()

    if resp.status_code == 206:  # Partial content
        mode = "ab"
        print(f"  Resuming {local_filename} from byte {resume_bytes}")
    else:
        mode = "wb"
        resume_bytes = 0

    total_size = int(resp.headers.get("content-length", 0)) + resume_bytes
    item = DownloadItem(url=url, filename=local_filename, size_bytes=total_size)

    # Download with progress
    with open(output_path, mode) as f:
        if tqdm:
            with tqdm(
                total=total_size,
                unit="B",
                unit_scale=True,
                desc=local_filename[:30],
                initial=resume_bytes,
            ) as pbar:
                for chunk in resp.iter_content(chunk_size=8192):
                    if chunk:
                        f.write(chunk)
                        pbar.update(len(chunk))
        else:
            for chunk in resp.iter_content(chunk_size=8192):
                if chunk:
                    f.write(chunk)

    # Verify hash
    if expected_hash:
        file_hash = sha256_file(output_path)
        if file_hash != expected_hash:
            raise ValueError(f"Hash mismatch: expected {expected_hash}, got {file_hash}")

    item.status = "completed"
    return item

def sha256_file(path: Path) -> str:
    h = hashlib.sha256()
    with open(path, "rb") as f:
        for chunk in iter(lambda: f.read(65536), b""):
            h.update(chunk)
    return h.hexdigest()

# --- CPU-bound post processing (multiprocessing) ---
def verify_hash(item: DownloadItem, output_dir: Path) -> DownloadItem:
    """Verify file integrity (CPU-bound due to hashing large files)"""
    if item.status != "completed":
        return item
    path = output_dir / item.filename
    if path.exists():
        start = time.perf_counter()
        h = sha256_file(path)
        elapsed = time.perf_counter() - start
        item.sha256 = h
        print(f"  Verified {item.filename} ({elapsed:.2f}s)")
    return item

# --- Main download manager ---
class ConcurrentDownloader:
    def __init__(
        self,
        output_dir: str = "downloads",
        max_thread_workers: int = 5,
        max_process_workers: int = 2,
    ):
        self.output_dir = Path(output_dir)
        self.output_dir.mkdir(exist_ok=True)
        self.max_thread_workers = max_thread_workers
        self.max_process_workers = max_process_workers

    def download_all(self, urls: list[str]) -> list[DownloadItem]:
        print(f"Downloading {len(urls)} files with {self.max_thread_workers} threads...")
        items = []

        with ThreadPoolExecutor(max_workers=self.max_thread_workers) as executor:
            futures = {
                executor.submit(download_file, url, self.output_dir): url
                for url in urls
            }

            for future in as_completed(futures):
                url = futures[future]
                try:
                    item = future.result()
                    items.append(item)
                except Exception as e:
                    items.append(DownloadItem(
                        url=url,
                        filename=url.split("/")[-1],
                        status="failed",
                        error=str(e),
                    ))
                    print(f"[FAIL] {url}: {e}")

        return items

    def verify_all(self, items: list[DownloadItem]) -> list[DownloadItem]:
        if not items:
            return items

        print(f"\nVerifying {len(items)} files with {self.max_process_workers} processes...")
        with ProcessPoolExecutor(max_workers=self.max_process_workers) as executor:
            futures = {
                executor.submit(verify_hash, item, self.output_dir): item
                for item in items
                if item.status == "completed"
            }

            for future in as_completed(futures):
                try:
                    future.result()  # Updates item in-place via mutation
                except Exception as e:
                    print(f"Verification error: {e}")

        return items

    def report(self, items: list[DownloadItem]):
        completed = sum(1 for i in items if i.status == "completed")
        failed = sum(1 for i in items if i.status == "failed")
        total_bytes = sum(i.size_bytes for i in items if i.status == "completed")

        print(f"\n{'='*40}")
        print(f"Download Report")
        print(f"{'='*40}")
        print(f"Total files: {len(items)}")
        print(f"Completed:   {completed}")
        print(f"Failed:      {failed}")
        print(f"Total data:  {total_bytes / 1024 / 1024:.1f} MB")
        print(f"Output dir:  {self.output_dir.resolve()}")

        if failed:
            print(f"\nFailed downloads:")
            for item in items:
                if item.status == "failed":
                    print(f"  - {item.url}: {item.error}")

# --- Demo ---
def main():
    sample_urls = [
        "https://httpbin.org/image/jpeg",
        "https://httpbin.org/image/png",
        "https://httpbin.org/image/webp",
        "https://httpbin.org/image/svg",
        "https://www.python.org/static/img/python-logo.png",
    ]

    downloader = ConcurrentDownloader(
        output_dir="downloads",
        max_thread_workers=3,
        max_process_workers=2,
    )

    start = time.perf_counter()

    # Phase 1: Download (threads — I/O bound)
    items = downloader.download_all(sample_urls)

    # Phase 2: Verify (processes — CPU bound hashing)
    items = downloader.verify_all(items)

    elapsed = time.perf_counter() - start

    downloader.report(items)
    print(f"Total time: {elapsed:.2f}s")

if __name__ == "__main__":
    main()
```

### Key concepts demonstrated
- `ThreadPoolExecutor` for concurrent I/O-bound downloads
- `ProcessPoolExecutor` for CPU-bound hash verification
- `as_completed()` for processing results as they finish
- Retry decorator for transient download failures
- HTTP Range header for resume support
- Streaming download with chunked iteration
- `tqdm` progress bars (optional)
- SHA-256 file hashing for integrity verification
- Dataclasses for structured result tracking
