## Banking System

A complete object-oriented banking system with savings/checking accounts, deposits, withdrawals, transfers, and transaction history.

```python
from abc import ABC, abstractmethod
from datetime import datetime
from typing import List, Optional


class Transaction:
    """Represents a single financial transaction."""

    def __init__(self, amount: float, transaction_type: str, description: str = ""):
        self.amount = amount
        self.transaction_type = transaction_type
        self.timestamp = datetime.now()
        self.description = description

    def __str__(self) -> str:
        return (f"[{self.timestamp.strftime('%Y-%m-%d %H:%M:%S')}] "
                f"{self.transaction_type:10s} | "
                f"${self.amount:>8.2f} | {self.description}")

    def __repr__(self) -> str:
        return (f"Transaction({self.amount}, '{self.transaction_type}', "
                f"'{self.description}')")


class BankAccount(ABC):
    """Abstract base class for all bank accounts."""

    _last_account_number = 1000

    def __init__(self, owner: str, initial_balance: float = 0.0):
        self.owner = owner
        self._balance = initial_balance
        self._account_number = BankAccount._generate_account_number()
        self._transactions: List[Transaction] = []
        self._is_active = True

        if initial_balance > 0:
            self._log_transaction(initial_balance, "DEPOSIT", "Initial deposit")

    @classmethod
    def _generate_account_number(cls) -> int:
        cls._last_account_number += 1
        return cls._last_account_number

    @property
    def balance(self) -> float:
        return self._balance

    @property
    def account_number(self) -> int:
        return self._account_number

    @property
    def transaction_count(self) -> int:
        return len(self._transactions)

    @abstractmethod
    def get_account_type(self) -> str:
        pass

    def deposit(self, amount: float, description: str = "Deposit") -> float:
        if amount <= 0:
            raise ValueError("Deposit amount must be positive")
        if not self._is_active:
            raise ValueError("Account is closed")

        self._balance += amount
        self._log_transaction(amount, "DEPOSIT", description)
        return self._balance

    def withdraw(self, amount: float, description: str = "Withdrawal") -> float:
        if amount <= 0:
            raise ValueError("Withdrawal amount must be positive")
        if not self._is_active:
            raise ValueError("Account is closed")
        if amount > self._balance:
            raise ValueError("Insufficient funds")

        self._balance -= amount
        self._log_transaction(amount, "WITHDRAWAL", description)
        return self._balance

    def transfer(self, amount: float, target_account: 'BankAccount',
                 description: str = "") -> None:
        if amount <= 0:
            raise ValueError("Transfer amount must be positive")
        if not description:
            description = f"Transfer to {target_account.owner}"

        self.withdraw(amount, f"Transfer out: {target_account.owner}")
        target_account.deposit(amount, f"Transfer in: {self.owner}")

        self._log_transaction(amount, "TRANSFER OUT",
                              f"To: {target_account.owner}")

    def get_transaction_history(self) -> List[Transaction]:
        return list(self._transactions)

    def close_account(self) -> None:
        self._is_active = False
        if self._balance > 0:
            self.withdraw(self._balance, "Account closing payout")
        self._log_transaction(0, "CLOSE", "Account closed")

    def _log_transaction(self, amount: float, txn_type: str, desc: str) -> None:
        txn = Transaction(amount, txn_type, desc)
        self._transactions.append(txn)

    def print_statement(self) -> None:
        print(f"\n{'='*60}")
        print(f"ACCOUNT STATEMENT")
        print(f"{'='*60}")
        print(f"Owner: {self.owner}")
        print(f"Account: {self.account_number} ({self.get_account_type()})")
        print(f"Status: {'Active' if self._is_active else 'Closed'}")
        print(f"{'='*60}")
        for txn in self._transactions:
            print(txn)
        print(f"{'='*60}")
        print(f"Current Balance: ${self._balance:,.2f}")
        print(f"{'='*60}\n")

    def __str__(self) -> str:
        return (f"[{self.get_account_type()}] {self.owner} - "
                f"#{self.account_number} - ${self._balance:,.2f}")

    def __repr__(self) -> str:
        return (f"{self.__class__.__name__}('{self.owner}', "
                f"{self._balance})")


class SavingsAccount(BankAccount):
    """Savings account with interest rate and minimum balance."""

    def __init__(self, owner: str, initial_balance: float = 0.0,
                 interest_rate: float = 0.02, min_balance: float = 0.0):
        super().__init__(owner, initial_balance)
        self._interest_rate = interest_rate
        self._min_balance = min_balance

    @property
    def interest_rate(self) -> float:
        return self._interest_rate

    def get_account_type(self) -> str:
        return "Savings"

    def apply_interest(self) -> float:
        """Apply monthly interest to the account."""
        if not self._is_active:
            raise ValueError("Account is closed")
        if self._balance > 0:
            interest = self._balance * (self._interest_rate / 12)
            self._balance += interest
            self._log_transaction(interest, "INTEREST",
                                  f"Monthly interest ({self._interest_rate*100:.1f}%)")
            return interest
        return 0.0

    def withdraw(self, amount: float, description: str = "Withdrawal") -> float:
        if self._balance - amount < self._min_balance:
            raise ValueError(f"Withdrawal would violate minimum balance of "
                             f"${self._min_balance:.2f}")
        return super().withdraw(amount, description)


class CheckingAccount(BankAccount):
    """Checking account with overdraft protection."""

    def __init__(self, owner: str, initial_balance: float = 0.0,
                 overdraft_limit: float = 500.0):
        super().__init__(owner, initial_balance)
        self._overdraft_limit = overdraft_limit

    @property
    def overdraft_limit(self) -> float:
        return self._overdraft_limit

    def get_account_type(self) -> str:
        return "Checking"

    def withdraw(self, amount: float, description: str = "Withdrawal") -> float:
        max_withdrawal = self._balance + self._overdraft_limit
        if amount > max_withdrawal:
            raise ValueError(f"Exceeds overdraft limit of ${self._overdraft_limit:.2f}")
        return super().withdraw(amount, description)


class Bank:
    """Manages multiple accounts."""

    def __init__(self, name: str):
        self.name = name
        self._accounts: List[BankAccount] = []

    def open_savings_account(self, owner: str, initial_deposit: float = 0.0,
                              interest_rate: float = 0.02,
                              min_balance: float = 0.0) -> SavingsAccount:
        account = SavingsAccount(owner, initial_deposit, interest_rate, min_balance)
        self._accounts.append(account)
        return account

    def open_checking_account(self, owner: str, initial_deposit: float = 0.0,
                               overdraft_limit: float = 500.0) -> CheckingAccount:
        account = CheckingAccount(owner, initial_deposit, overdraft_limit)
        self._accounts.append(account)
        return account

    def find_account(self, account_number: int) -> Optional[BankAccount]:
        for account in self._accounts:
            if account.account_number == account_number:
                return account
        return None

    def get_total_deposits(self) -> float:
        return sum(a.balance for a in self._accounts if a._is_active)

    def list_accounts(self) -> None:
        print(f"\n{'='*60}")
        print(f"{self.name}  -  Account List")
        print(f"{'='*60}")
        for account in self._accounts:
            print(f"  {account}")
        print(f"Total Deposits: ${self.get_total_deposits():,.2f}")
        print(f"{'='*60}\n")

def demo_banking_system():
    print("\n" + "="*60)
    print("BANKING SYSTEM DEMO")
    print("="*60)

    # Create a bank
    bank = Bank("Python National Bank")

    # Open accounts
    alice_savings = bank.open_savings_account("Alice", 5000, interest_rate=0.03)
    bob_checking = bank.open_checking_account("Bob", 1000, overdraft_limit=1000)
    charlie_savings = bank.open_savings_account("Charlie", 10000, min_balance=500)

    # List all accounts
    bank.list_accounts()

    # Perform transactions
    print("--- Transactions ---")
    alice_savings.deposit(1500, "Bonus")
    alice_savings.withdraw(200)
    print(f"Alice's balance: ${alice_savings.balance:,.2f}")

    bob_checking.withdraw(1500)  # Uses $500 overdraft
    print(f"Bob's balance: ${bob_checking.balance:,.2f}")

    # Transfer between accounts
    print("\n--- Transfer ---")
    charlie_savings.transfer(2000, alice_savings, "Birthday gift")
    print(f"Charlie: ${charlie_savings.balance:,.2f}")
    print(f"Alice: ${alice_savings.balance:,.2f}")

    # Apply interest
    print("\n--- Interest ---")
    interest = alice_savings.apply_interest()
    print(f"Alice earned ${interest:.2f} in interest")

    # Print statements
    alice_savings.print_statement()
    bob_checking.print_statement()

    # Close an account
    print("\n--- Close Account ---")
    alice_savings.close_account()
    print(f"Alice's account closed. Final balance: ${alice_savings.balance:,.2f}")

    # Error handling
    print("\n--- Error Handling ---")
    try:
        bob_checking.withdraw(100000)
    except ValueError as e:
        print(f"Error: {e}")

    bank.list_accounts()


if __name__ == "__main__":
    demo_banking_system()
```

## Inventory Management System

A complete inventory management system with products, suppliers, stock management, and reorder alerts.

```python
from datetime import datetime, timedelta
from typing import List, Optional, Dict
from enum import Enum
import json


class Category(Enum):
    ELECTRONICS = "Electronics"
    CLOTHING = "Clothing"
    FOOD = "Food"
    FURNITURE = "Furniture"
    BOOKS = "Books"
    OTHER = "Other"


class Supplier:
    """Represents a product supplier."""

    def __init__(self, supplier_id: int, name: str, contact: str,
                 phone: str, email: str, lead_time_days: int = 7):
        self.supplier_id = supplier_id
        self.name = name
        self.contact = contact
        self.phone = phone
        self.email = email
        self.lead_time_days = lead_time_days

    def __str__(self) -> str:
        return f"{self.name} (ID: {self.supplier_id})"

    def __repr__(self) -> str:
        return f"Supplier({self.supplier_id}, '{self.name}')"


class Product:
    """Represents a product in the inventory."""

    def __init__(self, product_id: int, name: str, category: Category,
                 unit_price: float, supplier: Supplier,
                 reorder_level: int = 10, reorder_quantity: int = 50):
        self.product_id = product_id
        self.name = name
        self.category = category
        self._unit_price = unit_price
        self.supplier = supplier
        self._quantity = 0
        self.reorder_level = reorder_level
        self.reorder_quantity = reorder_quantity
        self._last_restocked = None

    @property
    def unit_price(self) -> float:
        return self._unit_price

    @unit_price.setter
    def unit_price(self, value: float) -> None:
        if value < 0:
            raise ValueError("Price cannot be negative")
        self._unit_price = value

    @property
    def quantity(self) -> int:
        return self._quantity

    @property
    def total_value(self) -> float:
        return self._quantity * self._unit_price

    @property
    def needs_reorder(self) -> bool:
        return self._quantity <= self.reorder_level

    @property
    def last_restocked(self) -> Optional[datetime]:
        return self._last_restocked

    def add_stock(self, quantity: int) -> None:
        if quantity <= 0:
            raise ValueError("Quantity must be positive")
        self._quantity += quantity
        self._last_restocked = datetime.now()

    def remove_stock(self, quantity: int) -> int:
        if quantity <= 0:
            raise ValueError("Quantity must be positive")
        if quantity > self._quantity:
            raise ValueError(f"Insufficient stock. Available: {self._quantity}")
        self._quantity -= quantity
        return self._quantity

    def __str__(self) -> str:
        status = " REORDER" if self.needs_reorder else " OK"
        return (f"[{status}] {self.name:25s} | "
                f"Qty: {self._quantity:4d} | "
                f"${self._unit_price:>6.2f} | "
                f"Value: ${self.total_value:>8.2f}")

    def __repr__(self) -> str:
        return f"Product({self.product_id}, '{self.name}')"


class Inventory:
    """Manages the complete inventory."""

    def __init__(self):
        self._products: Dict[int, Product] = {}
        self._suppliers: Dict[int, Supplier] = {}
        self._next_product_id = 1
        self._next_supplier_id = 1

    def add_supplier(self, name: str, contact: str, phone: str,
                     email: str, lead_time_days: int = 7) -> Supplier:
        supplier = Supplier(self._next_supplier_id, name, contact,
                            phone, email, lead_time_days)
        self._suppliers[supplier.supplier_id] = supplier
        self._next_supplier_id += 1
        return supplier

    def add_product(self, name: str, category: Category,
                    unit_price: float, supplier: Supplier,
                    reorder_level: int = 10,
                    reorder_quantity: int = 50) -> Product:
        product = Product(self._next_product_id, name, category,
                          unit_price, supplier, reorder_level, reorder_quantity)
        self._products[product.product_id] = product
        self._next_product_id += 1
        return product

    def find_product(self, product_id: int) -> Optional[Product]:
        return self._products.get(product_id)

    def find_products_by_name(self, name: str) -> List[Product]:
        return [p for p in self._products.values()
                if name.lower() in p.name.lower()]

    def find_products_by_category(self, category: Category) -> List[Product]:
        return [p for p in self._products.values()
                if p.category == category]

    def receive_stock(self, product_id: int, quantity: int) -> Product:
        product = self.find_product(product_id)
        if not product:
            raise ValueError(f"Product {product_id} not found")
        product.add_stock(quantity)
        return product

    def sell_product(self, product_id: int, quantity: int) -> Product:
        product = self.find_product(product_id)
        if not product:
            raise ValueError(f"Product {product_id} not found")
        product.remove_stock(quantity)
        return product

    def get_reorder_list(self) -> List[Product]:
        return [p for p in self._products.values() if p.needs_reorder]

    def get_total_inventory_value(self) -> float:
        return sum(p.total_value for p in self._products.values())

    def get_category_summary(self) -> Dict[Category, Dict]:
        summary = {}
        for product in self._products.values():
            if product.category not in summary:
                summary[product.category] = {
                    "count": 0, "total_value": 0.0, "total_quantity": 0
                }
            summary[product.category]["count"] += 1
            summary[product.category]["total_value"] += product.total_value
            summary[product.category]["total_quantity"] += product.quantity
        return summary

    def generate_reorder_report(self) -> None:
        reorder_items = self.get_reorder_list()
        print(f"\n{'='*60}")
        print("REORDER REPORT  -  Items Below Reorder Level")
        print(f"{'='*60}")
        if not reorder_items:
            print("No items need reordering.")
        else:
            for product in sorted(reorder_items,
                                  key=lambda p: p.quantity):
                print(f"  {product}")
                print(f"    → Reorder {product.reorder_quantity} units "
                      f"from {product.supplier.name}")
        print(f"{'='*60}\n")

    def print_inventory(self) -> None:
        print(f"\n{'='*70}")
        print(f"FULL INVENTORY REPORT")
        print(f"{'='*70}")
        if not self._products:
            print("Inventory is empty.")
            return
        for product in self._products.values():
            print(f"  {product}")
        print(f"{'='*70}")
        print(f"Total Products: {len(self._products)}")
        print(f"Total Value: ${self.get_total_inventory_value():,.2f}")
        print(f"Items Needing Reorder: {len(self.get_reorder_list())}")
        print(f"{'='*70}\n")

def demo_inventory_system():
    print("\n" + "="*60)
    print("INVENTORY MANAGEMENT SYSTEM DEMO")
    print("="*60)

    inv = Inventory()

    # Add suppliers

    # Add products
    laptop = inv.add_product("Laptop Pro X1", Category.ELECTRONICS,

    mouse = inv.add_product("Wireless Mouse", Category.ELECTRONICS,

                             19.99, fash_wholesale, reorder_level=30, reorder_quantity=100)

                            59.99, fash_wholesale, reorder_level=15, reorder_quantity=50)

    # Receive stock
    inv.receive_stock(laptop.product_id, 25)
    inv.receive_stock(mouse.product_id, 150)

if __name__ == "__main__":
```

## Employee Management System

A complete employee management system with Employee, Manager, and Developer classes, payroll calculation, and department management.

```python
from abc import ABC, abstractmethod
from datetime import datetime, date
from typing import List, Dict, Optional, Set
from enum import Enum


class Department(Enum):
    ENGINEERING = "Engineering"
    MARKETING = "Marketing"
    SALES = "Sales"
    HR = "Human Resources"
    FINANCE = "Finance"
    DESIGN = "Design"


class PayrollError(Exception):
    """Custom exception for payroll errors."""
    pass


class Employee(ABC):
    """Abstract base class for all employees."""

    _id_counter = 0

    def __init__(self, name: str, email: str, department: Department,
                 hire_date: date, base_salary: float):
        Employee._id_counter += 1
        self._employee_id = Employee._id_counter
        self.name = name
        self.email = email
        self.department = department
        self.hire_date = hire_date
        self._base_salary = base_salary
        self._is_active = True
        self._performance_rating: Optional[float] = None
        self._projects: List[str] = []

    @property
    def employee_id(self) -> int:
        return self._employee_id

    @property
    def base_salary(self) -> float:
        return self._base_salary

    @base_salary.setter
    def base_salary(self, value: float) -> None:
        if value < 0:
            raise ValueError("Salary cannot be negative")
        self._base_salary = value

    @property
    def tenure_years(self) -> float:
        days_employed = (date.today() - self.hire_date).days
        return round(days_employed / 365.25, 1)

    @property
    def performance_rating(self) -> Optional[float]:
        return self._performance_rating

    @abstractmethod
    def calculate_pay(self) -> float:
        """Calculate the employee's total pay."""
        pass

    @abstractmethod
    def get_role(self) -> str:
        """Return the employee's role title."""
        pass

    def add_project(self, project_name: str) -> None:
        self._projects.append(project_name)

    def get_projects(self) -> List[str]:
        return list(self._projects)

    def set_performance_rating(self, rating: float) -> None:
        if rating < 0 or rating > 5:
            raise ValueError("Rating must be between 0 and 5")
        self._performance_rating = rating

    def terminate(self) -> None:
        self._is_active = False

    def __str__(self) -> str:
        status = "Active" if self._is_active else "Inactive"
        return (f"[{self.get_role():12s}] {self.name:20s} | "
                f"Dept: {self.department.value:15s} | "
                f"${self._base_salary:>8,.2f} | {status}")

    def __repr__(self) -> str:
        return (f"{self.__class__.__name__}('{self.name}', "
                f"'{self.email}', {self.department}, {self._base_salary})")


class Developer(Employee):
    """Developer with additional skills and project bonuses."""

    def __init__(self, name: str, email: str, department: Department,
                 hire_date: date, base_salary: float,
                 programming_languages: Optional[List[str]] = None):
        super().__init__(name, email, department, hire_date, base_salary)
        self._programming_languages = programming_languages or []
        self._completed_projects_bonus = 0.0

    def add_programming_language(self, language: str) -> None:
        self._programming_languages.append(language)

    def get_programming_languages(self) -> List[str]:
        return list(self._programming_languages)

    def complete_project(self, bonus: float = 0.0) -> None:
        self._completed_projects_bonus += bonus

    def get_role(self) -> str:
        return "Developer"

    def calculate_pay(self) -> float:
        pay = self._base_salary
        pay += self._completed_projects_bonus

        # Tenure bonus: 2% per year
        pay += self._base_salary * 0.02 * self.tenure_years

        # Performance bonus
        if self._performance_rating and self._performance_rating >= 4.0:
            pay += self._base_salary * 0.10

        return pay


class Manager(Employee):
    """Manager with team oversight and department budget."""

    def __init__(self, name: str, email: str, department: Department,
                 hire_date: date, base_salary: float,
                 team_size: int = 0):
        super().__init__(name, email, department, hire_date, base_salary)
        self._team_size = team_size
        self._department_budget = 0.0
        self._team_members: List[Employee] = []

    @property
    def team_size(self) -> int:
        return self._team_size

    @team_size.setter
    def team_size(self, value: int) -> None:
        if value < 0:
            raise ValueError("Team size cannot be negative")
        self._team_size = value

    def add_team_member(self, employee: Employee) -> None:
        self._team_members.append(employee)
        self._team_size = len(self._team_members)

    def get_team_members(self) -> List[Employee]:
        return list(self._team_members)

    def set_department_budget(self, budget: float) -> None:
        self._department_budget = budget

    def get_role(self) -> str:
        return "Manager"

    def calculate_pay(self) -> float:
        pay = self._base_salary

        # Management bonus: 5% of base
        pay += self._base_salary * 0.05

        # Team size bonus: $1000 per team member
        pay += self._team_size * 1000

        # Tenure bonus: 3% per year
        pay += self._base_salary * 0.03 * self.tenure_years

        return pay


class Payroll:
    """Handles payroll processing for all employees."""

    def __init__(self, company_name: str):
        self.company_name = company_name
        self._employees: Dict[int, Employee] = {}
        self._pay_periods: List[Dict] = []

    def add_employee(self, employee: Employee) -> None:
        self._employees[employee.employee_id] = employee
        print(f"Added: {employee}")

    def remove_employee(self, employee_id: int) -> None:
        if employee_id in self._employees:
            emp = self._employees[employee_id]
            emp.terminate()
            print(f"Terminated: {emp.name}")

    def find_employee(self, employee_id: int) -> Optional[Employee]:
        return self._employees.get(employee_id)

    def find_by_department(self, department: Department) -> List[Employee]:
        return [e for e in self._employees.values()
                if e.department == department and e._is_active]

    def find_by_name(self, name: str) -> List[Employee]:
        return [e for e in self._employees.values()
                if name.lower() in e.name.lower() and e._is_active]

    def run_payroll(self, period_label: str) -> Dict:
        print(f"\n{'='*60}")
        print(f"PAYROLL RUN: {period_label}")
        print(f"{'='*60}")

        period_payroll = {
            "period": period_label,
            "date": datetime.now(),
            "payments": {}
        }
        total_payroll = 0.0

        for emp in self._employees.values():
            if not emp._is_active:
                continue

            pay = emp.calculate_pay()
            total_payroll += pay
            period_payroll["payments"][emp.employee_id] = {
                "name": emp.name,
                "role": emp.get_role(),
                "pay": pay
            }

            print(f"  ${pay:>8,.2f} | {emp.get_role():12s} | {emp.name}")

        print(f"{'='*60}")
        print(f"Total Payroll: ${total_payroll:>10,.2f}")
        print(f"Total Employees: {sum(1 for e in self._employees.values() if e._is_active)}")
        print(f"{'='*60}\n")

        self._pay_periods.append(period_payroll)
        return period_payroll

    def generate_department_report(self) -> None:
        print(f"\n{'='*60}")
        print("DEPARTMENT REPORT")
        print(f"{'='*60}")

        dept_data: Dict[Department, Dict] = {}
        for emp in self._employees.values():
            if not emp._is_active:
                continue
            if emp.department not in dept_data:
                dept_data[emp.department] = {
                    "count": 0,
                    "total_salary": 0.0,
                    "employees": []
                }
            dept_data[emp.department]["count"] += 1
            dept_data[emp.department]["total_salary"] += emp.base_salary
            dept_data[emp.department]["employees"].append(emp.name)

        for dept, data in sorted(dept_data.items(), key=lambda x: x[1]["count"],
                                  reverse=True):
            print(f"  {dept.value:15s} | "
                  f"Employees: {data['count']:2d} | "
                  f"Avg Salary: ${data['total_salary']/data['count']:>8,.2f}")
            for name in data["employees"]:
                print(f"    - {name}")

    def generate_employee_report(self) -> None:
        print(f"\n{'='*70}")
        print("EMPLOYEE ROSTER")
        print(f"{'='*70}")
        for emp in sorted(self._employees.values(),
                          key=lambda e: e.employee_id):
            status = "Active" if emp._is_active else "Inactive"
            print(f"  #{emp.employee_id:3d} | {emp}")
            if emp.performance_rating:
                print(f"       Rating: {emp.performance_rating}/5.0 | "
                      f"Tenure: {emp.tenure_years} yrs | "
                      f"Pay: ${emp.calculate_pay():,.2f}")
            if emp.get_projects():
                print(f"       Projects: {', '.join(emp.get_projects())}")
        print(f"{'=':->70}")
```

## OOP Principles Applied (Summary Across Projects)

| Principle | Banking System | Inventory Management | Employee Management |
|-----------|---------------|---------------------|-------------------|
| **Classes & Objects** | `BankAccount`, `SavingsAccount`, `Transaction` | `Product`, `Supplier`, `Inventory` | `Employee`, `Manager`, `Developer` |
| **Constructors** | `__init__` initializes balance, account number, transactions | `__init__` sets product attributes, supplier info | `__init__` sets salary, department, hire date |
| **Encapsulation** | Private `_balance`, `_transactions` with `@property` getters | Private `_quantity`, `_unit_price` with controlled access | Private `_base_salary`, `_employee_id` with properties |
| **Inheritance** | `SavingsAccount` and `CheckingAccount` extend `BankAccount` |  -  (`Product` stands alone with `Supplier` association) | `Developer` and `Manager` extend `Employee` |
| **Polymorphism** | `withdraw()` behaves differently for savings vs checking |  -  | `calculate_pay()` differs for Manager vs Developer |
| **Abstraction** | `BankAccount` is ABC with `get_account_type()` abstract |  -  | `Employee` is ABC with `calculate_pay()` and `get_role()` abstract |
| **Magic Methods** | `__str__`, `__repr__` for account display | `__str__`, `__repr__` for products | `__str__`, `__repr__` for employee display |

## Key Takeaways

- **Banking System**: Best demonstrates encapsulation (protecting balance) and polymorphism (different withdrawal rules)
- **Inventory Management**: Best demonstrates object composition (Product references Supplier), property decorators, and business logic (reorder alerts)
- **Employee Management**: Best demonstrates abstract base classes, method overriding, and complex payroll calculations
