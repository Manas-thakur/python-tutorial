```python
"""Data models for the Student Management System."""
from dataclasses import dataclass, field
from typing import List
from datetime import datetime


@dataclass
class Student:
    """Represents a student with their academic information."""
    name: str
    age: int
    grade: str
    subjects: List[str] = field(default_factory=list)
    student_id: int = field(default=0)
    email: str = ""
    phone: str = ""
    enrollment_date: str = field(default_factory=lambda: datetime.now().isoformat()[:10])

    def to_dict(self):
        """Convert to dictionary for JSON serialization."""
        return {
            "student_id": self.student_id,
            "name": self.name,
            "age": self.age,
            "grade": self.grade,
            "subjects": self.subjects,
            "email": self.email,
            "phone": self.phone,
            "enrollment_date": self.enrollment_date
        }

    @classmethod
    def from_dict(cls, data):
        """Create Student from dictionary."""
        return cls(
            name=data["name"],
            age=data["age"],
            grade=data["grade"],
            subjects=data.get("subjects", []),
            student_id=data["student_id"],
            email=data.get("email", ""),
            phone=data.get("phone", ""),
            enrollment_date=data.get("enrollment_date", "")
        )
```

```python
"""Logging configuration for the application."""
import logging
import logging.handlers
import sys
from pathlib import Path


def setup_logger(name="student_manager"):
    """Configure application logger."""
    logger = logging.getLogger(name)
    logger.setLevel(logging.DEBUG)
    logger.handlers.clear()

    log_dir = Path("logs")
    log_dir.mkdir(exist_ok=True)

    formatter = logging.Formatter(
        '%(asctime)s | %(levelname)-8s | %(message)s',
        datefmt='%Y-%m-%d %H:%M:%S'
    )

    console = logging.StreamHandler(sys.stdout)
    console.setLevel(logging.INFO)
    console.setFormatter(formatter)
    logger.addHandler(console)

    file_handler = logging.handlers.RotatingFileHandler(
        log_dir / "app.log",
        maxBytes=2 * 1024 * 1024,
        backupCount=3
    )
    file_handler.setLevel(logging.DEBUG)
    file_handler.setFormatter(formatter)
    logger.addHandler(file_handler)

    return logger
```

```python
# --- student_manager.py ---
"""Core Student Manager with CRUD operations."""
import json
from pathlib import Path
from typing import List, Optional, Dict, Any
from models import Student
from logger_config import setup_logger


class StudentManager:
    """Manages student records with JSON persistence."""

    def __init__(self, data_file="data/students.json"):
        self.data_file = Path(data_file)
        self.data_file.parent.mkdir(parents=True, exist_ok=True)
        self.logger = setup_logger()
        self.students: List[Student] = []
        self._next_id = 1
        self.load()

    # --- Persistence ---
    def load(self):
        """Load students from JSON file."""
        if self.data_file.exists():
            try:
                with open(self.data_file, "r") as f:
                    data = json.load(f)
                self.students = [Student.from_dict(s) for s in data.get("students", [])]
                self._next_id = data.get("next_id", len(self.students) + 1)
                self.logger.info("Loaded %d students from %s", len(self.students), self.data_file)
            except (json.JSONDecodeError, FileNotFoundError) as e:
                self.logger.error("Failed to load data: %s", e)
                self.students = []
                self._next_id = 1
        else:
            self.logger.info("No data file found, starting fresh")
            self.students = []

    def save(self):
        """Save students to JSON file."""
        try:
            data = {
                "next_id": self._next_id,
                "students": [s.to_dict() for s in self.students],
                "updated_at": __import__('datetime').datetime.now().isoformat()
            }
            with open(self.data_file, "w") as f:
                json.dump(data, f, indent=2)
            self.logger.debug("Saved %d students", len(self.students))
            return True
        except (IOError, PermissionError) as e:
            self.logger.error("Failed to save data: %s", e)
            return False

    # --- CREATE ---
    def add_student(self, name: str, age: int, grade: str,
                    subjects=None, email="", phone="") -> Student:
        """Add a new student."""
        if not name or not name.strip():
            raise ValueError("Name cannot be empty")
        if age < 5 or age > 120:
            raise ValueError("Age must be between 5 and 120")
        if not grade:
            raise ValueError("Grade cannot be empty")

        # Check for duplicate
        if any(s.name.lower() == name.lower() for s in self.students):
            self.logger.warning("Duplicate student name: %s", name)
            raise ValueError(f"Student '{name}' already exists")

        student = Student(
            student_id=self._next_id,
            name=name.strip(),
            age=age,
            grade=grade.upper(),
            subjects=subjects or [],
            email=email,
            phone=phone
        )
        self._next_id += 1
        self.students.append(student)
        self.save()
        self.logger.info("Added student: %s (ID: %d)", student.name, student.student_id)
        return student

    # --- READ ---
    def get_all(self, sort_by="name") -> List[Student]:
        """Get all students, sorted."""
        if sort_by == "name":
            return sorted(self.students, key=lambda s: s.name.lower())
        elif sort_by == "age":
            return sorted(self.students, key=lambda s: s.age)
        elif sort_by == "grade":
            return sorted(self.students, key=lambda s: s.grade)
        elif sort_by == "id":
            return sorted(self.students, key=lambda s: s.student_id)
        return self.students

    def search(self, query: str = "", field: str = "all") -> List[Student]:
        """Search students by name, grade, or subject."""
        if not query:
            return self.students

        query = query.lower()
        results = []

        for s in self.students:
            if field == "name" and query in s.name.lower():
                results.append(s)
            elif field == "grade" and query in s.grade.lower():
                results.append(s)
            elif field == "subject" and any(query in sub.lower() for sub in s.subjects):
                results.append(s)
            elif field == "all":
                if (query in s.name.lower() or
                    query in s.grade.lower() or
                    any(query in sub.lower() for sub in s.subjects)):
                    results.append(s)

        return results

    def get_by_id(self, student_id: int) -> Optional[Student]:
        """Get a student by ID."""
        for s in self.students:
            if s.student_id == student_id:
                return s
        return None

    # --- UPDATE ---
    def update_student(self, student_id: int, **updates) -> Student:
        """Update student fields."""
        student = self.get_by_id(student_id)
        if not student:
            raise ValueError(f"Student with ID {student_id} not found")

        allowed = {"name", "age", "grade", "subjects", "email", "phone"}
        for key, value in updates.items():
            if key in allowed:
                if key == "name" and value:
                    value = value.strip()
                    if any(s.name.lower() == value.lower() and s.student_id != student_id
                           for s in self.students):
                        raise ValueError(f"Name '{value}' already exists")
                if key == "age" and (value < 5 or value > 120):
                    raise ValueError("Age must be between 5 and 120")
                setattr(student, key, value)

        self.save()
        self.logger.info("Updated student ID %d: %s", student_id, student.name)
        return student

    # --- DELETE ---
    def delete_student(self, student_id: int) -> bool:
        """Delete a student by ID."""
        student = self.get_by_id(student_id)
        if not student:
            raise ValueError(f"Student with ID {student_id} not found")

        self.students = [s for s in self.students if s.student_id != student_id]
        self.save()
        self.logger.info("Deleted student: %s (ID: %d)", student.name, student_id)
        return True

    # --- STATISTICS ---
    def get_stats(self) -> Dict[str, Any]:
        """Generate statistics about the student body."""
        if not self.students:
            return {"total": 0}

        grades = {}
        age_groups = {"<18": 0, "18-25": 0, ">25": 0}
        all_subjects = []

        for s in self.students:
            grades[s.grade] = grades.get(s.grade, 0) + 1
            if s.age < 18:
                age_groups["<18"] += 1
            elif s.age <= 25:
                age_groups["18-25"] += 1
            else:
                age_groups[">25"] += 1
            all_subjects.extend(s.subjects)

        subject_counts = {}
        for sub in all_subjects:
            subject_counts[sub] = subject_counts.get(sub, 0) + 1

        return {
            "total": len(self.students),
            "grades": dict(sorted(grades.items())),
            "age_groups": age_groups,
            "average_age": round(sum(s.age for s in self.students) / len(self.students), 1),
            "most_popular_subjects": dict(
                sorted(subject_counts.items(), key=lambda x: x[1], reverse=True)[:5]
            ),
            "students_per_grade": grades
        }

    def display_students(self, students=None):
        """Display students in a formatted table."""
        students = students or self.students
        if not students:
            print("\n📋 No students found.")
            return

        print(f"\n{'ID':<5} {'Name':<25} {'Age':<5} {'Grade':<8} {'Subjects':<30} {'Email':<25}")
        print("─" * 100)
        for s in students:
            subs = ", ".join(s.subjects[:3])
            if len(s.subjects) > 3:
                subs += "..."
            print(f"{s.student_id:<5} {s.name:<25} {s.age:<5} {s.grade:<8} {subs:<30} {s.email:<25}")
        print(f"\nTotal: {len(students)} students")
```

```python
"""
Library Management System
A complete system for managing books, members, and borrowing operations.
"""

import json
from pathlib import Path
from datetime import datetime, timedelta
from typing import List, Optional, Dict, Any
import logging

logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s | %(levelname)-8s | %(message)s',
    handlers=[
        logging.FileHandler('library.log'),
        logging.StreamHandler()
    ]
)
logger = logging.getLogger("library")


# ============================================================
# DATA MODELS
# ============================================================

class Book:
    """Represents a book in the library."""

    def __init__(self, title: str, author: str, isbn: str = "",
                 year: int = 0, copies: int = 1, book_id: int = 0):
        self.book_id = book_id
        self.title = title
        self.author = author
        self.isbn = isbn
        self.year = year
        self.copies = copies
        self.available_copies = copies

    def to_dict(self):
        return {
            "book_id": self.book_id,
            "title": self.title,
            "author": self.author,
            "isbn": self.isbn,
            "year": self.year,
            "copies": self.copies,
            "available_copies": self.available_copies
        }

    @classmethod
    def from_dict(cls, data):
        book = cls(
            title=data["title"],
            author=data["author"],
            isbn=data.get("isbn", ""),
            year=data.get("year", 0),
            copies=data.get("copies", 1),
            book_id=data["book_id"]
        )
        book.available_copies = data.get("available_copies", book.copies)
        return book

    def is_available(self) -> bool:
        return self.available_copies > 0

    def __str__(self):
        status = "Available" if self.is_available() else "Checked out"
        return f"[{self.book_id}] {self.title} by {self.author} ({status})"


class Member:
    """Represents a library member."""

    def __init__(self, name: str, email: str = "",
                 phone: str = "", member_id: int = 0):
        self.member_id = member_id
        self.name = name
        self.email = email
        self.phone = phone
        self.borrowed_books: List[int] = []
        self.join_date = datetime.now().isoformat()[:10]

    def to_dict(self):

            "name": self.name,

            "phone": self.phone,

        }

            phone=data.get("phone", ""),
            member_id=data["member_id"]

        member.borrowed_books = data.get("borrowed_books", [])
        member.join_date = data.get("join_date", "")
        return member


class Transaction:
    """Represents a borrow/return transaction."""

    def __init__(self, book_id: int, member_id: int,
                 borrow_date=None, due_date=None,
                 return_date=None, transaction_id: int = 0):
        self.transaction_id = transaction_id
        self.book_id = book_id
        self.member_id = member_id
        self.borrow_date = borrow_date or datetime.now().isoformat()[:10]
        self.due_date = due_date or (
            datetime.now() + timedelta(days=14)
        ).isoformat()[:10]
        self.return_date = return_date
        self.fine = 0.0

    def is_overdue(self) -> bool:
        if self.return_date:
            return False
        return datetime.now().date() > datetime.strptime(
            self.due_date, "%Y-%m-%d"
        ).date()

    def calculate_fine(self, rate_per_day=1.0) -> float:
        if self.return_date:
            return 0.0
        if not self.is_overdue():
            return 0.0
        due = datetime.strptime(self.due_date, "%Y-%m-%d")
        days_overdue = (datetime.now() - due).days
        return round(days_overdue * rate_per_day, 2)

    def to_dict(self):
        return {
            "transaction_id": self.transaction_id,
            "book_id": self.book_id,
            "member_id": self.member_id,
            "borrow_date": self.borrow_date,
            "due_date": self.due_date,
            "return_date": self.return_date,
            "fine": self.fine
        }

    @classmethod
    def from_dict(cls, data):
        t = cls(
            book_id=data["book_id"],
            member_id=data["member_id"],
            borrow_date=data.get("borrow_date"),
            due_date=data.get("due_date"),
            return_date=data.get("return_date"),
            transaction_id=data["transaction_id"]
        )
        t.fine = data.get("fine", 0.0)
        return t

# LIBRARY SYSTEM
# ============================================================

class Library:
    """Core library management system."""

    def __init__(self, data_dir="library_data"):
        self.data_dir = Path(data_dir)
        self.data_dir.mkdir(exist_ok=True)

        self.books: List[Book] = []
        self.members: List[Member] = []
        self.transactions: List[Transaction] = []

        self._next_book_id = 1
        self._next_member_id = 1
        self._next_transaction_id = 1

        self._load()

    # --- Persistence ---
    def _save(self):
        """Save all data to JSON files."""
        data = {
            "books": {
                "next_id": self._next_book_id,
                "items": [b.to_dict() for b in self.books]
            },
            "members": {
                "next_id": self._next_member_id,
                "items": [m.to_dict() for m in self.members]
            },
            "transactions": {
                "next_id": self._next_transaction_id,
                "items": [t.to_dict() for t in self.transactions]
            }
        }
        with open(self.data_dir / "data.json", "w") as f:
            json.dump(data, f, indent=2)
        logger.debug("Data saved successfully")

    def _load(self):
        """Load all data from JSON files."""
        data_file = self.data_dir / "data.json"
        if not data_file.exists():
            logger.info("No existing data found, starting fresh")
            return

        try:
            with open(data_file) as f:
                data = json.load(f)

            books_data = data.get("books", {})
            self._next_book_id = books_data.get("next_id", 1)
            self.books = [Book.from_dict(b) for b in books_data.get("items", [])]

            members_data = data.get("members", {})
            self._next_member_id = members_data.get("next_id", 1)
            self.members = [Member.from_dict(m) for m in members_data.get("items", [])]

            trans_data = data.get("transactions", {})
            self._next_transaction_id = trans_data.get("next_id", 1)
            self.transactions = [Transaction.from_dict(t)
                                 for t in trans_data.get("items", [])]

            logger.info("Loaded %d books, %d members, %d transactions",
                       len(self.books), len(self.members), len(self.transactions))
        except (json.JSONDecodeError, FileNotFoundError) as e:
            logger.error("Error loading data: %s", e)

    # --- Book Operations ---
    def add_book(self, title, author, isbn="", year=0, copies=1) -> Book:
        """Add a new book to the library."""
        if not title or not author:
            raise ValueError("Title and author are required")

        book = Book(title, author, isbn, year, copies, self._next_book_id)
        self._next_book_id += 1
        self.books.append(book)
        self._save()
        logger.info("Added book: %s by %s", title, author)
        return book

    def find_books(self, query="", field="all") -> List[Book]:
        """Find books by title, author, or ISBN."""
        query = query.lower()
        results = []
        for book in self.books:
            if field == "title" and query in book.title.lower():
                results.append(book)
            elif field == "author" and query in book.author.lower():
                results.append(book)
            elif field == "isbn" and query in book.isbn.lower():
                results.append(book)
            elif field == "all":
                if (query in book.title.lower() or
                    query in book.author.lower() or
                    query in book.isbn.lower()):
                    results.append(book)
        return results

    def get_book(self, book_id) -> Optional[Book]:
        for book in self.books:
            if book.book_id == book_id:
                return book
        return None

    def remove_book(self, book_id) -> bool:
        book = self.get_book(book_id)
        if not book:
            raise ValueError(f"Book ID {book_id} not found")
        self.books.remove(book)
        self._save()
        logger.info("Removed book: %s", book.title)
        return True

    # --- Member Operations ---
    def add_member(self, name, email="", phone="") -> Member:
        if not name:
            raise ValueError("Member name is required")
        member = Member(name, email, phone, self._next_member_id)
        self._next_member_id += 1
        self.members.append(member)
        self._save()
        logger.info("Added member: %s (ID: %d)", name, member.member_id)
        return member

    def find_members(self, query="") -> List[Member]:
        query = query.lower()
        return [m for m in self.members
                if query in m.name.lower() or query in m.email.lower()]

    def get_member(self, member_id) -> Optional[Member]:
        for m in self.members:
            if m.member_id == member_id:
                return m
        return None

    def remove_member(self, member_id) -> bool:
        member = self.get_member(member_id)
        if not member:
            raise ValueError(f"Member ID {member_id} not found")
        if member.borrowed_books:
            raise ValueError(f"Cannot remove member with {len(member.borrowed_books)} borrowed books")
        self.members.remove(member)
        self._save()
        logger.info("Removed member: %s", member.name)
        return True

    # --- Borrow/Return Operations ---
    def borrow_book(self, member_id: int, book_id: int,
                    due_days: int = 14) -> Transaction:
        """Borrow a book. Returns a Transaction."""
        member = self.get_member(member_id)
        if not member:
            raise ValueError(f"Member ID {member_id} not found")

        book = self.get_book(book_id)
        if not book:
            raise ValueError(f"Book ID {book_id} not found")

        if not book.is_available():
            raise ValueError(f"'{book.title}' has no available copies")

        # Check for overdue books
        for tid in member.borrowed_books:
            trans = self._get_transaction(tid)
            if trans and trans.is_overdue():
                fine = trans.calculate_fine()
                logger.warning("Member %s has overdue book (fine: $%.2f)",
                             member.name, fine)

        # Create transaction
        borrow_date = datetime.now()
        due_date = borrow_date + timedelta(days=due_days)
        transaction = Transaction(
            book_id=book_id,
            member_id=member_id,
            borrow_date=borrow_date.isoformat()[:10],
            due_date=due_date.isoformat()[:10],
            transaction_id=self._next_transaction_id
        )
        self._next_transaction_id += 1

        # Update records
        book.available_copies -= 1
        member.borrowed_books.append(transaction.transaction_id)
        self.transactions.append(transaction)
        self._save()

        logger.info("Book '%s' borrowed by %s, due %s",
                   book.title, member.name, transaction.due_date)
        return transaction

    def return_book(self, member_id: int, book_id: int) -> Transaction:
        """Return a borrowed book."""
        member = self.get_member(member_id)
        if not member:
            raise ValueError(f"Member ID {member_id} not found")

        book = self.get_book(book_id)
        if not book:
            raise ValueError(f"Book ID {book_id} not found")

        # Find active transaction
        transaction = None
        for tid in member.borrowed_books:
            t = self._get_transaction(tid)
            if t and t.book_id == book_id and t.return_date is None:
                transaction = t
                break

        if not transaction:
            raise ValueError(f"No active borrow record for book ID {book_id}")

        # Process return
        transaction.return_date = datetime.now().isoformat()[:10]
        transaction.fine = transaction.calculate_fine()
        book.available_copies += 1
        member.borrowed_books.remove(transaction.transaction_id)
        self._save()

        if transaction.fine > 0:
            logger.info("Book '%s' returned by %s (fine: $%.2f)",
                       book.title, member.name, transaction.fine)
        else:
            logger.info("Book '%s' returned by %s (on time)",
                       book.title, member.name)
        return transaction

    def _get_transaction(self, tid: int) -> Optional[Transaction]:
        for t in self.transactions:
            if t.transaction_id == tid:
                return t
        return None

    def get_overdue_books(self) -> List[Transaction]:
        """Get all overdue transactions."""
        return [t for t in self.transactions
                if t.return_date is None and t.is_overdue()]

    def list_borrowed_by(self, member_id: int) -> List[Transaction]:
        """Get all active borrows for a member."""
        member = self.get_member(member_id)
        if not member:
            return []
        result = []
        for tid in member.borrowed_books:
            t = self._get_transaction(tid)
            if t:
                result.append(t)
        return result

    def get_member_fines(self, member_id: int) -> float:
        """Calculate total fines for a member."""
        total = 0.0
        for t in self.transactions:
            if t.member_id == member_id and t.return_date is None and t.is_overdue():
                total += t.calculate_fine()
        return round(total, 2)

    # --- Reports ---
    def get_report(self) -> Dict[str, Any]:
        """Generate a library report."""
        available = sum(b.available_copies for b in self.books)
        borrowed = sum(b.copies - b.available_copies for b in self.books)
        overdue = len(self.get_overdue_books())

        return {
            "total_books": len(self.books),
            "total_copies": sum(b.copies for b in self.books),
            "available_copies": available,
            "borrowed_copies": borrowed,
            "total_members": len(self.members),
            "active_members": len([m for m in self.members if m.borrowed_books]),
            "overdue_transactions": overdue,
            "total_transactions": len(self.transactions)
        }

    def __str__(self):
        report = self.get_report()
        return (f"Library: {report['total_books']} books, "
                f"{report['total_members']} members, "
                f"{report['borrowed_copies']} borrowed, "
                f"{report['overdue_transactions']} overdue")
```

## Project 3: File Organizer

A tool that scans directories and organizes files by extension, date, or size, using pathlib for all file operations.

### Features
- Organize by file extension (e.g., all .pdf → PDF/ folder)
- Organize by modification date (e.g., 2024/01-Jan/)
- Organize by file size (small/medium/large)
- Dry-run mode to preview before moving
- Undo capability (dry-run logs everything)
- Recursive or flat organization
- Logging of all operations

### Code

```python
# --- file_organizer.py ---
"""
File Organizer Tool
Organize files by extension, date, or size using pathlib.
"""

from pathlib import Path
from datetime import datetime
import logging
import sys
import json
import shutil

logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s | %(levelname)-8s | %(message)s',
    datefmt='%Y-%m-%d %H:%M:%S',
    handlers=[
        logging.StreamHandler(sys.stdout),
        logging.FileHandler('organizer.log')
    ]
)
logger = logging.getLogger("organizer")


class FileOrganizer:
    """Organize files in a directory by various criteria."""

    def __init__(self, base_dir=".", dry_run=True):
        self.base_dir = Path(base_dir).resolve()
        if not self.base_dir.exists():
            raise FileNotFoundError(f"Directory not found: {base_dir}")
        self.dry_run = dry_run
        self.operations_log = []
        self.organized_count = 0
        self.skipped_count = 0
        self.error_count = 0

    def organize_by_extension(self, recursive=True, move=True):
        """Organize files into folders named by extension."""
        method = "Moving" if move else "Copying"
        mode = "DRY RUN" if self.dry_run else "EXECUTING"
        logger.info("%s: Organize by extension (%s) [%s]", mode, method, self.base_dir)
        print(f"\n{'─' * 60}")
        print(f"  {mode}: Organize by Extension")
        print(f"  Directory: {self.base_dir}")
        print(f"{'─' * 60}")

        pattern = "**/*" if recursive else "*"
        files = [f for f in self.base_dir.glob(pattern) if f.is_file()]

        for file_path in files:
            # Skip dotfiles and organizer script
            if file_path.name.startswith('.') or file_path == Path(__file__).resolve():
                self.skipped_count += 1
                continue

            ext = file_path.suffix.lower() if file_path.suffix else "NO_EXTENSION"
            ext_name = ext[1:].upper() if ext.startswith('.') else ext
            target_dir = self.base_dir / ext_name

            self._process_file(file_path, target_dir, move)

        self._print_summary("Extension")

    def organize_by_date(self, date_field="modified", recursive=True):
        """Organize files into Year/Month folders by date."""
        mode = "DRY RUN" if self.dry_run else "EXECUTING"
        logger.info("Organize by %s date [%s]", date_field, mode)
        print(f"\n{'─' * 60}")
        print(f"  {mode}: Organize by {date_field.capitalize()} Date")
        print(f"  Directory: {self.base_dir}")
        print(f"{'─' * 60}")

        pattern = "**/*" if recursive else "*"
        files = [f for f in self.base_dir.glob(pattern) if f.is_file()]

        for file_path in files:
            if file_path.name.startswith('.'):
                self.skipped_count += 1
                continue

            try:
                stat = file_path.stat()
                if date_field == "modified":
                    timestamp = stat.st_mtime
                elif date_field == "created":
                    timestamp = stat.st_ctime
                elif date_field == "accessed":
                    timestamp = stat.st_atime
                else:
                    raise ValueError(f"Unknown date field: {date_field}")

                dt = datetime.fromtimestamp(timestamp)
                year = str(dt.year)
                month = f"{dt.month:02d}-{dt.strftime('%b')}"
                target_dir = self.base_dir / year / month

                self._process_file(file_path, target_dir, move=True)

            except (OSError, ValueError) as e:
                logger.error("Error processing %s: %s", file_path.name, e)
                self.error_count += 1

        self._print_summary("Date")

    def organize_by_size(self, recursive=True):
        """Organize files by size categories: Small, Medium, Large, Huge."""
        mode = "DRY RUN" if self.dry_run else "EXECUTING"
        logger.info("Organize by size [%s]", mode)
        print(f"\n{'─' * 60}")
        print(f"  {mode}: Organize by File Size")
        print(f"  Directory: {self.base_dir}")
        print(f"{'─' * 60}")

        # Size categories in bytes
        categories = {
            "Tiny": (0, 1024),                 # < 1 KB
            "Small": (1024, 100 * 1024),       # 1 KB - 100 KB
            "Medium": (100 * 1024, 1024 * 1024),  # 100 KB - 1 MB
            "Large": (1024 * 1024, 100 * 1024 * 1024),  # 1 MB - 100 MB
            "Huge": (100 * 1024 * 1024, float('inf'))    # > 100 MB
        }

        pattern = "**/*" if recursive else "*"
        files = [f for f in self.base_dir.glob(pattern) if f.is_file()]

        for file_path in files:
            if file_path.name.startswith('.'):
                self.skipped_count += 1
                continue

            try:
                size = file_path.stat().st_size
                for cat_name, (low, high) in categories.items():
                    if low <= size < high:
                        target_dir = self.base_dir / cat_name
                        self._process_file(file_path, target_dir, move=True)
                        break
            except OSError as e:
                logger.error("Error processing %s: %s", file_path.name, e)
                self.error_count += 1

        self._print_summary("Size")

    def _process_file(self, file_path: Path, target_dir: Path, move: bool):
        """Process a single file: move/copy to target directory."""
        try:
            target_dir.mkdir(parents=True, exist_ok=True)
            dest = target_dir / file_path.name

            if dest.exists():
                # Add number suffix to avoid overwrite
                stem = file_path.stem
                suffix = file_path.suffix
                counter = 1
                while dest.exists():
                    dest = target_dir / f"{stem}_{counter}{suffix}"
                    counter += 1

            rel_path = file_path.relative_to(self.base_dir)
            rel_dest = dest.relative_to(self.base_dir)

            if self.dry_run:
                action = "Would move" if move else "Would copy"
                print(f"  📄  {rel_path}")
                print(f"       → {rel_dest}")
                self.operations_log.append({
                    "source": str(rel_path),
                    "destination": str(rel_dest),
                    "action": "move" if move else "copy"
                })
            else:
                if move:
                    file_path.rename(dest)
                    logger.debug("Moved: %s → %s", rel_path, rel_dest)
                else:
                    shutil.copy2(file_path, dest)
                    logger.debug("Copied: %s → %s", rel_path, rel_dest)
                action = "Moved" if move else "Copied"
                print(f"  {action}: {rel_path} → {rel_dest}")

            self.organized_count += 1

        except (OSError, shutil.Error) as e:
            logger.error("Failed to process %s: %s", file_path.name, e)
            self.error_count += 1

    def _print_summary(self, method: str):
        """Print organization summary."""
        mode = "(DRY RUN)" if self.dry_run else ""
        print(f"\n{'─' * 60}")
        print(f"  Summary - Organize by {method} {mode}")
        print(f"  Organized: {self.organized_count}")
        print(f"  Skipped:   {self.skipped_count}")
        print(f"  Errors:    {self.error_count}")
        print(f"{'─' * 60}")

        if not self.dry_run:
            logger.info("Organization complete: %d organized, %d skipped, %d errors",
                       self.organized_count, self.skipped_count, self.error_count)

    def preview(self):
        """Preview all operations that would be performed."""
        print(f"\n{'=' * 60}")
        print(f"  PREVIEW: {len(self.operations_log)} operations planned")
        print(f"{'=' * 60}")

        # Group by destination directory
        dirs = {}
        for op in self.operations_log:
            dest_dir = Path(op["destination"]).parent
            dirs.setdefault(str(dest_dir), []).append(op)

        for dest_dir, ops in sorted(dirs.items()):
            print(f"\n  📁 {dest_dir}/ ({len(ops)} files)")
            for op in ops:
                print(f"       {Path(op['source']).name}")

        print(f"\n  Total: {len(self.operations_log)} files")
        return self.operations_log

    def save_plan(self, filename="organize_plan.json"):
        """Save the operations plan to a JSON file."""
        with open(filename, "w") as f:
            json.dump(self.operations_log, f, indent=2)
        logger.info("Plan saved to %s", filename)
        print(f"  Plan saved to: {filename}")

    def undo_last(self):
        """Undo the last organization by reversing moves."""
        if not self.operations_log:
            print("No operations to undo.")
            return

        print(f"\n{'─' * 60}")
        print("  UNDO: Reversing last organization")
        print(f"{'─' * 60}")

        undone = 0
        for op in reversed(self.operations_log):
            source = self.base_dir / op["destination"]
            dest = self.base_dir / op["source"]

            if source.exists():
                dest.parent.mkdir(parents=True, exist_ok=True)
                source.rename(dest)
                print(f"  Restored: {op['destination']} → {op['source']}")
                undone += 1
            else:
                print(f"  ⚠️  Not found: {op['destination']}")

        print(f"\n  Undone: {undone} operations")
        self.operations_log = []
        logger.info("Undo complete: %d files restored", undone)


# --- Interactive CLI ---
def main():
    """Interactive file organizer CLI."""
    import argparse

    parser = argparse.ArgumentParser(description="File Organizer Tool")
    parser.add_argument("directory", nargs="?", default=".",
                       help="Directory to organize (default: current)")
    parser.add_argument("--method", "-m", choices=["ext", "date", "size"],
                       default="ext", help="Organization method")
    parser.add_argument("--date-field", "-d", choices=["modified", "created", "accessed"],
                       default="modified", help="Date field for date-based sorting")
    parser.add_argument("--no-recursive", action="store_true",
                       help="Don't recurse into subdirectories")
    parser.add_argument("--execute", "-e", action="store_true",
                       help="Actually move files (default is dry-run)")

    args = parser.parse_args()

    organizer = FileOrganizer(args.directory, dry_run=not args.execute)

    try:
        if args.method == "ext":
            organizer.organize_by_extension(recursive=not args.no_recursive)
        elif args.method == "date":
            organizer.organize_by_date(
                date_field=args.date_field,
                recursive=not args.no_recursive
            )
        elif args.method == "size":
            organizer.organize_by_size(recursive=not args.no_recursive)

        if not args.execute:
            # In dry-run mode, show summary and offer to execute
            print(f"\n{'=' * 60}")
            response = input("  Preview looks good? Execute organization? (yes/no): ")
            if response.lower() == 'yes':
                organizer.dry_run = False
                organizer.organized_count = 0
                organizer.skipped_count = 0
                organizer.error_count = 0
                organizer.operations_log = []

                if args.method == "ext":
                    organizer.organize_by_extension(recursive=not args.no_recursive, move=True)
                elif args.method == "date":
                    organizer.organize_by_date(
                        date_field=args.date_field,
                        recursive=not args.no_recursive
                    )
                elif args.method == "size":
                    organizer.organize_by_size(recursive=not args.no_recursive)
            else:
                print("  Cancelled. No files were modified.")
                # Save the plan for reference
                organizer.save_plan()

    except KeyboardInterrupt:
        print("\n  Interrupted. No files were modified (dry-run mode).")
    except Exception as e:
        logger.error("Unexpected error: %s", e, exc_info=True)
        print(f"  Error: {e}")


if __name__ == "__main__":
    main()
```
